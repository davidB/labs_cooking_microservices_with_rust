Déclarer un struct


						struct Vec2 {
							x: f64,
							y: f64,
						}
					

						pub struct Vec2 {
							pub x: f64,
							pub y: f64,
						}
					

							fn main(){
								let v = Vec2{x: 10.0, y: 11_f64};
								println!("display v: {}", v);
							}

Ajouter une fonction


							impl Vec2 {
								fn to_string(&self) -> String {
									format!("Vec2 {{ x: {}, y: {} }}", self.x, self.y)
								}
							}
							println!("display v: {}", v.to_string());
					

						impl Vec2 {
							fn distance(&self, other: &Vec2) -> f32 {
								let dx = self.x - other.x;
								let dy = self.y - other.y;
								((dx.pow(2) + dy.pow(2)) as f32).sqrt()
							}
						}

						let a = Vec2 { x: 0, y: 0 };
						let b = Vec2 { x: 1, y: 1 };
						
						println!("{}", a.distance(&b));

Implémenter un trait


						impl Add for Vec2 {
							type Output = Vec2;
							fn add(self, rhs: Vec2) -> Vec2 {
								Vec2 {
									x: self.x + rhs.x,
									y: self.y + rhs.y,
								}
							}
						}
						
						impl AddAssign for Vec2 {
							fn add_assign(&mut self, rhs: Vec2) {
								self.x += rhs.x;
								self.y += rhs.y;
							}
						}

Implémenter un trait


						use std::fmt;

						impl fmt::Debug for Vec2 {
							fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
								write!(f, "Vec2 {{ x: {:?}, y: {:?} }}", self.x, self.y)
							}
						}

						println!("debug v {:?}", v);

Dérivation automatique


						#[derive(Debug, Display, Clone, Copy, Hash, PartialEq, Eq)]

Générique


					fn twice(x: T) -> U
					where T: Add<Output=U>,
						T: Copy
					{
						x + x
					}

					fn main() {
						let one = Vec2 { x: 1, y: 1 };
						
						println!("{:?}", twice(one));
					}

Macros


						macro_rules! print_twice {
							( $x:expr ) => {
								{
									println!("{:?}", twice($x));
								}
							};
						}
						print_twice!(Vec2 { x: 1, y: 1 });
