# #![allow(unused_variables)]

use actix_web::http;

mod reviews;
#}

# #![allow(unused_variables)]
#fn main() {
server::new(move || {
    App::with_state(AppState {
    }).middleware(Logger::default())
        .resource("/reviews/{product_id}", |r| {
            r.method(http::Method::GET).with2(reviews::reviews);
            r.method(http::Method::POST).with3(reviews::create_review);
        })
}).bind(addr)
    .unwrap()
    .start();
#}
